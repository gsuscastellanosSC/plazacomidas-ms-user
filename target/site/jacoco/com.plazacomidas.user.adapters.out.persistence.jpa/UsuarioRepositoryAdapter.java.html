package com.plazacomidas.user.adapters.out.persistence.jpa;

import com.plazacomidas.user.adapters.out.persistence.jpa.mapper.UserEntityMapper;
import com.plazacomidas.user.adapters.out.persistence.jpa.repository.UsuarioJpaRepository;
import com.plazacomidas.user.adapters.out.persistence.jpa.helper.RoleAdapterHelper;
import com.plazacomidas.user.application.mapper.UsuarioMapper;
import com.plazacomidas.user.domain.exception.ResourceNotFoundException;
import com.plazacomidas.user.domain.model.User;
import com.plazacomidas.user.domain.model.UserEntity;
import com.plazacomidas.user.domain.port.out.UserPersistencePort;
import com.plazacomidas.user.domain.util.RoleConstants;
import com.plazacomidas.user.domain.util.UserConstants;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

@Component
@RequiredArgsConstructor
public class UsuarioRepositoryAdapter implements UserPersistencePort {

    private final UsuarioJpaRepository usuarioJpaRepository;
    private final RoleAdapterHelper rolAdapterHelper;

    @Override
    public User save(User usuario) {
        Long idRol = rolAdapterHelper.getIdByName(RoleConstants.ROLE_OWNER);

        UserEntity entity = UserEntityMapper.toEntity(usuario, idRol);

        UserEntity saved = usuarioJpaRepository.save(entity);
        usuario.setId(saved.getId());
        return usuario;
    }

    @Override
    public User getById(Long id) {

        UserEntity entity = usuarioJpaRepository.findById(id)
                .orElseThrow(() -> new ResourceNotFoundException(UserConstants.USER_NOT_FOUND));

        String nombreRol = rolAdapterHelper.getNameById(entity.getRoleId());

        return UsuarioMapper.toDomain(entity, nombreRol);
    }
}
